PROJECT(qucsfilter CXX C)
CMAKE_MINIMUM_REQUIRED(VERSION 2.6)
cmake_policy(VERSION 2.6)

SET(BITMAPDIR TRUE)
SET(BINARYDIR TRUE)
SET(DOCDIR TRUE)
SET(LANGUAGEDIR TRUE)
SET(LIBRARYDIR TRUE)

SET(CMAKE_BUILD_TYPE Debug)

ADD_DEFINITIONS( -DHAVE_CONFIG_H )

# configure the header config.h
CONFIGURE_FILE (
    "${PROJECT_SOURCE_DIR}/../config.h.cmake"
    "${PROJECT_BINARY_DIR}/config.h"
)

INCLUDE_DIRECTORIES("${PROJECT_BINARY_DIR}")

FIND_PACKAGE( Qt4 REQUIRED )
SET( QT_USE_QTGUI TRUE )
#SET( QT_USE_QT3SUPPORT TRUE )

INCLUDE( ${QT_USE_FILE} )

ADD_DEFINITIONS(${QT_DEFINITIONS})

SET(QUCS-FILTER_SRCS
helpdialog.cpp
lc_filter.cpp
main.cpp
qf_cauer.cpp
qf_filter.cpp
qf_poly.cpp
qucsfilter.cpp
)

SET(QUCS-FILTER_MOC_HDRS
helpdialog.h
qucsfilter.h
)

QT4_WRAP_CPP(QUCS-FILTER_MOC_SRCS ${QUCS-FILTER_MOC_HDRS})

ADD_EXECUTABLE(qucs-filter ${QUCS-FILTER_SRCS} ${QUCS-FILTER_MOC_SRCS})

TARGET_LINK_LIBRARIES(qucs-filter ${QT_LIBRARIES})

INSTALL(TARGETS qucs-filter DESTINATION bin)
